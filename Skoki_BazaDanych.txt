CREATE TABLE Kraj (
    id_kraju NUMERIC(6) PRIMARY KEY,
    nazwa VARCHAR(20) NOT NULL
);
CREATE TABLE Konkurs (
    id_konkursu NUMERIC(6) PRIMARY KEY,
    termin_zgloszen date
);
CREATE TABLE Zawodnik (
    id_zawodnika NUMERIC(6) PRIMARY KEY,
    imie VARCHAR(20) NOT NULL,
    nazwisko VARCHAR(20) NOT NULL,
    id_kraju NUMERIC(6) NOT NULL REFERENCES Kraj,
    punkty NUMERIC(6)
);
CREATE TABLE KwotaStartowa (
    id_kwoty NUMERIC(6) PRIMARY KEY,
    id_kraju NUMERIC(6) NOT NULL REFERENCES Kraj,
    id_konkursu NUMERIC(6) NOT NULL REFERENCES Konkurs,
    UNIQUE (id_kraju, id_konkursu),
    kwota_startowa NUMERIC(6) NOT NULL
);
CREATE TABLE Zgloszenie (
    id_zgloszenia NUMERIC(6) PRIMARY KEY,
    id_konkursu NUMERIC(6) NOT NULL REFERENCES Konkurs,
    id_zawodnika NUMERIC(6) NOT NULL REFERENCES Zawodnik,
    numer_startowy NUMERIC(6) NOT NULL
);
CREATE TABLE Skok (
    id_skoku NUMERIC(6) PRIMARY KEY,
    odleglosc NUMERIC(6) NOT NULL,
    ocena NUMERIC(6) NOT NULL,
    numer_serii NUMERIC(6) NOT NULL,
    id_zgloszenia NUMERIC(6) REFERENCES Zgloszenie,
    UNIQUE(numer_serii, id_zgloszenia)
);
